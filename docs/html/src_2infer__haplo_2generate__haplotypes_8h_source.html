62inline HaploList generateHaplotypePairs(const std::vector<int>& genotype) {
63 HaploList result;
64 std::function<void(size_t, Haplotype, Haplotype)> backtrack =
65 [&](size_t index, Haplotype h1, Haplotype h2) {
66 if (index == genotype.size()) {
67 result.emplace_back(h1, h2);
68 return;
69 }
70 if (genotype[index] == 0) {
71 h1.push_back(0);
72 h2.push_back(0);
73 backtrack(index + 1, h1, h2);
74 } else if (genotype[index] == 1) {
75 h1.push_back(1);
76 h2.push_back(1);
77 backtrack(index + 1, h1, h2);
78 } else if (genotype[index] == 2) {
79 Haplotype h1_copy = h1, h2_copy = h2;
80 h1.push_back(0);
81 h2.push_back(1);
82 backtrack(index + 1, h1, h2);
83 h1_copy.push_back(1);
84 h2_copy.push_back(0);
85 backtrack(index + 1, h1_copy, h2_copy);
86 }
87 };
88 backtrack(0, {}, {});
89 return result;
90}
